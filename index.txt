    <script>
        try {
            if (typeof d3 === 'undefined') {
                throw new Error("The D3.js library was blocked from loading. Please check your internet connection or disable strict tracking protection/ad-blockers for this local file.");
            }

            const treeData = {
                name: "Interpretability Methods",
                description: "The root classification framework for rendering tree-based ensembles transparent.",
                children: [
                    {
                        name: "Acquired", // CHANGED FROM POST-HOC
                        description: "Techniques applied after an opaque model has been trained to extract explanations.",

                            { name: "Extracted Rules", description: "Algorithmically distilling decision paths into logical rule sets.<br><br><b>Primary Domains:</b> Healthcare (Causal estimators), Finance & Law (Exact/MILP rules for legal compliance)." },
                            {
                                name: "Surrogate Models",
                                description: "Training a secondary, interpretable model to mimic the complex ensemble.",
                                children: [
                                    { name: "Tree", description: "Approximating the ensemble using a single decision tree.<br><br><b>Primary Domains:</b> Finance & Law (Local surrogates like LIME), General usage." },
                                    { name: "Non-tree", description: "Approximating using linear models or alternative structures.<br><br><b>Primary Domains:</b> General usage." }
                                ]
                            },
                            { name: "Regularization Based", description: "Applying mathematical constraints to simplify decision boundaries post-training.<br><br><b>Primary Domains:</b> General usage." },
                            {
                                name: "Modular",
                                description: "Decomposing the feature space to apply localized explanations.",
                                children: [
                                    { name: "Tree", description: "Using localized decision trees for specific sub-regions.<br><br><b>Primary Domains:</b> Engineering (High-speed, safety-critical inference)." },
                                    { name: "Non-tree", description: "Using alternative localized estimators for sub-regions.<br><br><b>Primary Domains:</b> Engineering (Sensor concept drift)." }
                                ]
                            },
                            {
                                name: "Visualization Oriented",
                                description: "Translating structural topology into visual abstractions.",
                                children: [
                                    { name: "Rule-based", description: "Visualizing the topology of extracted rules.<br><br><b>Primary Domains:</b> General usage." },
                                    { name: "Feature-based", description: "Visualizing feature contributions via heatmaps or attributions.<br><br><b>Primary Domains:</b> Healthcare (Biomarker validation), Engineering (Physical law validation)." },
                                    { name: "Point-based", description: "Explaining predictions through representative data prototypes.<br><br><b>Primary Domains:</b> General usage." }
                                ]
                            },
                            {
                                name: "Counterfactuals",
                                description: "Identifying minimum necessary changes to alter the model's output.",
                                children: [
                                    { name: "Feature-based", description: "Modifying individual feature values via mathematical optimization.<br><br><b>Primary Domains:</b> Fairness-Aware Auditing, General usage." },
                                    { name: "Set-based", description: "Defining a continuous region offering robustness guarantees.<br><br><b>Primary Domains:</b> Finance & Law (Strict regulatory recourse)." },
                                    { name: "Instance-based", description: "Grounding the explanation in actual historical training instances.<br><br><b>Primary Domains:</b> Healthcare (Actionable clinical interventions)." }
                                ]
                            }
                        ]
                    },
                    {
                        name: "Inherent",
                        description: "Architectures designed to be transparent by nature, shifting focus to ante-hoc design.",
                        children: [
                            { name: "Aggregational", description: "Summarizing or fusing the tree ensemble directly without secondary surrogate models.<br><br><b>Primary Domains:</b> General usage." },
                            {
                                name: "Modular",
                                description: "Building inherently interpretable sub-models optimized for different data regions.",
                                children: [
                                    { name: "Tree", description: "Using inherent local decision trees.<br><br><b>Primary Domains:</b> Engineering (Real-time risk assessment)." },
                                    { name: "Non-tree", description: "Using inherent non-tree local models.<br><br><b>Primary Domains:</b> Engineering (Time-series drift detection)." }
                                ]
                            },
                            { name: "Distributional", description: "Providing inherently transparent, probabilistic, or distributional outputs.<br><br><b>Primary Domains:</b> Privacy Preserving, Federated Learning (Industry 4.0, decentralized hospitals)." }
                        ]
                    }
                ]
            };

            // Clear the loading text
            document.getElementById("tree-container").innerHTML = "";

            const margin = {top: 20, right: 150, bottom: 20, left: 180},
                  width = 950 - margin.left - margin.right,
                  height = 550 - margin.top - margin.bottom;

            const svg = d3.select("#tree-container").append("svg")
                .attr("width", width + margin.right + margin.left)
                .attr("height", height + margin.top + margin.bottom)
                .append("g")
                .attr("transform", "translate(" + margin.left + "," + margin.top + ")");

            let i = 0;
            const root = d3.hierarchy(treeData, d => d.children);
            root.x0 = height / 2;
            root.y0 = 0;

            const treeMap = d3.tree().size([height, width]);

            root.children.forEach(collapse);
            update(root);

            function collapse(d) {
                if(d.children) {
                    d._children = d.children;
                    d._children.forEach(collapse);
                    d.children = null;
                }
            }
            
            function getNodeColor(d) {
                if (d._children) return "#3498db"; // Collapsed -> Solid Blue
                if (d.children) return "#fff";     // Expanded -> White
                return "#ecf0f1";                  // Leaf node -> Light Grey
            }

            function update(source) {
                const treeInfo = treeMap(root);
                const nodes = treeInfo.descendants(),
                      links = treeInfo.descendants().slice(1);

                nodes.forEach(d => d.y = d.depth * 200);

                const node = svg.selectAll('g.node')
                    .data(nodes, d => d.id || (d.id = ++i));

                const nodeEnter = node.enter().append('g')
                    .attr('class', 'node')
                    .attr("transform", d => "translate(" + source.y0 + "," + source.x0 + ")")
                    .on('click', click);

                nodeEnter.append('circle')
                    .attr('r', 1e-6)
                    .style("fill", getNodeColor);

                nodeEnter.append('text')
                    .attr("dy", ".35em")
                    .attr("x", d => d.children || d._children ? -13 : 13)
                    .attr("text-anchor", d => d.children || d._children ? "end" : "start")
                    .text(d => d.data.name);

                const nodeUpdate = nodeEnter.merge(node);
                nodeUpdate.transition().duration(400)
                    .attr("transform", d => "translate(" + d.y + "," + d.x + ")");

                nodeUpdate.select('circle')
                    .attr('r', 8)
                    .style("fill", getNodeColor)
                    .attr('cursor', 'pointer');

                const nodeExit = node.exit().transition().duration(400)
                    .attr("transform", d => "translate(" + source.y + "," + source.x + ")")
                    .remove();

                nodeExit.select('circle').attr('r', 1e-6);
                nodeExit.select('text').style('fill-opacity', 1e-6);

                const link = svg.selectAll('path.link')
                    .data(links, d => d.id);

                const linkEnter = link.enter().insert('path', "g")
                    .attr("class", "link")
                    .attr('d', d => {
                        const o = {x: source.x0, y: source.y0};
                        return diagonal(o, o);
                    });

                const linkUpdate = linkEnter.merge(link);
                linkUpdate.transition().duration(400)
                    .attr('d', d => diagonal(d, d.parent));

                link.exit().transition().duration(400)
                    .attr('d', d => {
                        const o = {x: source.x, y: source.y};
                        return diagonal(o, o);
                    })
                    .remove();

                nodes.forEach(d => {
                    d.x0 = d.x;
                    d.y0 = d.y;
                });

                function diagonal(s, d) {
                    return `M ${s.y} ${s.x}
                            C ${(s.y + d.y) / 2} ${s.x},
                              ${(s.y + d.y) / 2} ${d.x},
                              ${d.y} ${d.x}`;
                }
            }

            function click(event, d) {
                document.getElementById("info-title").innerText = d.data.name;
                // CHANGED TO innerHTML TO SUPPORT <br> and <b> TAGS
                document.getElementById("info-desc").innerHTML = d.data.description || "Sub-category exploring specific methodologies within the parent domain.";

                if (d.children) {
                    d._children = d.children;
                    d.children = null;
                } else {
                    d.children = d._children;
                    d._children = null;
                }
                update(d);
            }
        } catch (error) {
            document.getElementById("tree-container").innerHTML = 
                "<div style='text-align: center; color: #c0392b; padding: 20px;'>" +
                "<h3>Visualization Error</h3>" +
                "<p>" + error.message + "</p>" +
                "</div>";
        }
